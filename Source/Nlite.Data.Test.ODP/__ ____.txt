CREATE OR REPLACE TRIGGER orders_insert
 BEFORE
  INSERT
 ON orders
REFERENCING NEW AS NEW OLD AS OLD
 FOR EACH ROW
BEGIN
   SELECT nextid.NEXTVAL
     INTO :NEW.orderid
     FROM DUAL;
END;
/
